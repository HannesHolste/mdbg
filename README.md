## Install

```shell
npm install mdbg
```

## Usage

```js
const mdbg = require('mdbg')
mdbg.get('苹果')
// {
//   "traditional": "蘋果",
//   "simplified": "苹果",
//   "data": {
//     "ping2 guo3": {
//       "mandarin": "píng guó",
//       "definitions": [
//         "apple"
//       ],
//       "classifiers": {
//         "ge4": {
//           "traditional": "個",
//           "simplified": "个",
//           "mandarin": "gè",
//           "definition": "classifier for people or objects in general"
//         },
//         "ke1": {
//           "traditional": "顆",
//           "simplified": "颗",
//           "mandarin": "kē",
//           "definition": "classifier for small spheres, pearls, corn grains, teeth, hearts, satellites etc"
//         }
//       }
//     }
//   }
// }
mdbg.get('嗎')
// {
//   "traditional": "嗎",
//   "simplified": "吗",
//   "data": {
//     "ma3": {
//       "mandarin": "mǎ",
//       "definitions": [
//         "see 嗎啡|吗啡, morphine"
//       ]
//     },
//     "ma5": {
//       "mandarin": "ma",
//       "definitions": [
//         "(question particle for \"yes-no\" questions)"
//       ]
//     }
//   }
// }
```
